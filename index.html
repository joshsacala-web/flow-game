// 🧠 SMART MAZE (flow-based) function generate(){ grid = Array.from({length:rows},()=>Array(cols).fill(1)); let goalX = cols-2; let goalY = rows-2; function carve(x,y, lastDir=null){ // directions with bias toward goal let dirs = [ [1,0],[-1,0],[0,1],[0,-1] ]; dirs.sort((a,b)=>{ // bias toward goal let da = Math.abs((x+a[0]*2)-goalX)+Math.abs((y+a[1]*2)-goalY); let db = Math.abs((x+b[0]*2)-goalX)+Math.abs((y+b[1]*2)-goalY); // prefer same direction (flow) if(lastDir && a[0]===lastDir[0] && a[1]===lastDir[1]) da -= 2; if(lastDir && b[0]===lastDir[0] && b[1]===lastDir[1]) db -= 2; return da - db + (Math.random()*2); // slight randomness }); for(let [dx,dy] of dirs){ let nx = x + dx*2; let ny = y + dy*2; if(grid[ny] && grid[ny][nx]===1){ grid[ny][nx]=0; grid[y+dy][x+dx]=0; carve(nx,ny,[dx,dy]); } } } player = {x:1,y:1}; goal = {x:goalX,y:goalY}; grid[player.y][player.x]=0; carve(player.x,player.y); grid[goal.y][goal.x]=0; // ✨ CLEANUP: remove some dead ends for(let i=0;i<100;i++){ for(let y=1;y= 3){ let dirs = [[1,0],[-1,0],[0,1],[0,-1]]; let [dx,dy] = dirs[Math.floor(Math.random()*dirs.length)]; if(grid[y+dy] && grid[y+dy][x+dx]===1){ grid[y+dy][x+dx]=0; } } } } } } }
